import { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { authService } from '../api/auth.service';

export default function Login() {
  const [credentials, setCredentials] = useState({ email: '', password: '' });
  const [error, setError] = useState('');
  const navigate = useNavigate();

  const handleLogin = async (e) => {
    e.preventDefault();
    try {
      const response = await authService.login(credentials);
      localStorage.setItem('token', response.token);
      localStorage.setItem('user', JSON.stringify(response.user));
      navigate('/dashboard');
    } catch (err) {
      setError('Invalid credentials');
    }
  };

  return (
    <div className="login-container">
      <div className="login-card">
        <img src="/dga-logo.svg" alt="DGA Logo" />
        <h2>Digital Government Authority</h2>
        <h3>Oversight Platform</h3>
        <form onSubmit={handleLogin}>
          <input 
            type="email" 
            placeholder="Email"
            value={credentials.email}
            onChange={(e) => setCredentials({...credentials, email: e.target.value})}
          />
          <input 
            type="password" 
            placeholder="Password"
            value={credentials.password}
            onChange={(e) => setCredentials({...credentials, password: e.target.value})}
          />
          {error && <div className="error">{error}</div>}

      </div>
    </div>
  );
}

### A.3 Cross-Entity Integration Patterns

#### **Pattern 1: Identity Federation**
**Entities Involved**: MOI (Absher) as identity provider, all ministries as service providers

```
User ‚Üí Absher Login ‚Üí OAuth 2.0 Token ‚Üí Service (MOH/MOE/MOC/etc.)
```

**Benefits**:
- Single sign-on (SSO)
- Centralized identity management
- Audit trail
- User consent management

---

#### **Pattern 2: Data Exchange via Government Service Bus (GSB)**

**Example Flow: Business License Issuance**
```
1. User applies via MOC portal
2. MOC ‚Üí GSB ‚Üí MOI: Verify owner identity
3. MOC ‚Üí GSB ‚Üí ZATCA: Check tax registration
4. MOC ‚Üí GSB ‚Üí MHRSD: Verify labor quota
5. MOC ‚Üí GSB ‚Üí Municipality: Check zoning approval
6. MOC: Issue license (3 minutes total)
```

**GSB Technical Specs**:
- Protocol: REST APIs (JSON)
- Authentication: OAuth 2.0 + mTLS
- Message queue: Apache Kafka
- Average latency: 150ms per call
- Throughput: 10,000 TPS

---

#### **Pattern 3: Open Data Publishing**

**National Open Data Platform Architecture**:
```
Ministry Data Source ‚Üí ETL Pipeline ‚Üí Data Lake ‚Üí API Gateway ‚Üí Public/Developer Access
```
